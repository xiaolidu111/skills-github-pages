66.格式化日期

传入两个参数,一个是日期,一个是要转换成什么格式

yyyy替换成年

mm替换成月

dd替换成天


console.log(dateFormat(new Date('2020-12-01'), 'yyyy/MM/dd')); // 2020/12/01
console.log(dateFormat(new Date('2020-04-01'), 'yyyy/MM/dd')); // 2020/04/01
console.log(dateFormat(new Date('2020-04-01'), 'yyyy年MM月dd日'));// 2020年04月01日
function dateFormat (data, format) {
  let year = data.getFullYear()
  let month = data.getMonth() + 1
  let day = data.getDate()
  format = format.replace('yyyy', year)
  format = format.replace('MM', month)
  format = format.replace('dd', day)
  return format
}

  let args = [...arguments]

68.将数字从千分位用逗号隔开

先分成整数和小数部分

然后对整数部分进行分

如果证书长度小于3 直接拼接

如果整数长度大于3 就计算余数 然后后面用match进行匹配,再拼接 

let format = n => {
  let num = n.toString() // 转成字符串
  let decimals = ''
  // 判断是否有小数
  num.indexOf('.') > -1 ? decimals = num.split('.')[1] : decimals
  let len = num.split('.')[0].length
  if (len <= 3) {
    return num
  } else {
    let temp = ''
    let remainder = len % 3
    decimals ? temp = '.' + decimals : temp
    if (remainder > 0) { // 不是3的整数倍
      return num.slice(0, remainder) + ',' + num.slice(remainder, len).match(/\d{3}/g).join(',') + temp
    } else { // 是3的整数倍
      return num.slice(0, len).match(/\d{3}/g).join(',') + temp
    }
  }
}
console.log(format(1011.3334));  // '12,323.33'

69.循环打印红黄绿

69.1使用settimeout

创建一个打印事件的方法,接受参数为,时间,亮灯,和回调函数

定时器内执行亮灯,亮完之后执行回调函数

代码如下


function red () {
  console.log('red');
}
function green () {
  console.log('green');
}
function yellow () {
  console.log('yellow');
}

const task = (time, light, callback) => {
  setTimeout(() => {
    if (light === 'red') {
      red()
    } else if (light === 'green') {
      green()
    } else if (light === 'yellow') {
      yellow()
    }
    callback()
  }, time);
}

const step = () => {
  task(1000, 'red', () => {
    task(1000, 'green', () => {
      task(1000, 'yellow', step)
    })
  })
}
step()

70.实现发布订阅模式

设计一个类,里面有个对象,统计对应的类型,和函数 


class EventCenter {
  // 1. 定义事件容器，用来装事件数组
  handlers = {}

  // 2. 添加事件方法，参数：事件名 事件方法
  addEventListener (type, handler) {
    // 创建新数组容器
    if (!this.handlers[type]) {
      this.handlers[type] = []
    }
    // 存入事件
    this.handlers[type].push(handler)
  }

  // 3. 触发事件，参数：事件名 事件参数
  dispatchEvent (type, ...params) {
    // 若没有注册该事件则抛出错误
    if (!this.handlers[type]) {
      return new Error('该事件未注册')
    }
    // 触发事件
    this.handlers[type].forEach(handler => {
      handler(...params)
    })
  }

  // 4. 事件移除，参数：事件名 要删除事件，若无第二个参数则删除该事件的订阅和发布
  removeEventListener (type, handler) {
    if (!this.handlers[type]) {
      return new Error('事件无效')
    }
    if (!handler) {
      // 移除事件
      delete this.handlers[type]
    } else {
      const index = this.handlers[type].findIndex(el => el === handler)
      if (index === -1) {
        return new Error('无该绑定事件')
      }
      // 移除事件
      this.handlers[type].splice(index, 1)
      if (this.handlers[type].length === 0) {
        delete this.handlers[type]
      }
    }
  }
}

let fn = (...params) => {
  console.log('nihao');
  console.log(params);
}
moshi.addEventListener('你好', fn)
moshi.dispatchEvent('你好', 1)
moshi.removeEventListener('你好', fn)
