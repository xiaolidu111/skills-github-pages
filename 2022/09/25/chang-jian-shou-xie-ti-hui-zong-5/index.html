42.1 冒泡排序 

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArray = function (nums) {
  for (let i = 0; i < nums.length - 1; i++) { // 判断排序次数 一个是length-1次
    for (let j = 0; j < nums.length - 1 - i; j++) { // 内层循环 从0到已经排好序的地方
      if (nums[j] > nums[j + 1]) { // 有不同就交换
        [nums[j], nums[j + 1]] = [nums[j + 1], nums[j]]
      }
    }
  }
  return nums
};
let arr = [1, 8, 5, 9, 10, 3, 6, 2, 8, 21, 13]
console.log(sortArray(arr));
42.2选择排序 

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArray = function (nums) {
  for (let i = 0; i < nums.length; i++) {
    let minIndex = i
    for (let j = i + 1; j < nums.length; j++) {
      if (nums[j] < nums[minIndex]) {
        minIndex = j
      }
    }
    [nums[minIndex], nums[i]] = [nums[i], nums[minIndex]]
  }
  return nums
};
let arr = [1, 8, 5, 9, 10, 3, 6, 2, 8, 21, 13]

42.3 插入排序 
/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArray = function (nums) {
    for (let i = 1; i < nums.length; i++) {
        let preIndex = i - 1
        let current = nums[i]
        while (preIndex >= 0 && current < nums[preIndex]) {
            nums[preIndex + 1] = nums[preIndex]
            preIndex--
        }
        nums[preIndex + 1] = current
    }
    return nums
};
let arr = [1, 8, 9, 5, 10, 3, 6, 2, 8, 21, 13]

42.4归并排序 

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArray = function (nums) {
  // 归并排序需要两个函数 一个是将数组分成两部分 一个是将两部分的数组合并
  if (nums.length === 1) return nums // 如果数组的长度是1 就直接返回这个数组就可以了
  let mid = Math.floor(nums.length / 2) // 取出中间的位置 为了将数组分成两部分
  let left = nums.slice(0, mid) // 左边的部分
  let right = nums.slice(mid) // 右边的部分
  return merge(sortArray(left), sortArray(right)) // 递归的调用sort函数进行排序得到两个数组  然后将两个数组进行merge
};
function merge (left, right) {
  // merge函数 参数是两个数组 一个是左 一个是右
  let m = left && left.length // m 是左边数组的长度
  let n = right && right.length // n 是右边数组的长度
  let result = [] // 结果数组
  let i = 0 // 遍历左边数组的下标
  let j = 0 // 遍历右边数组的下标
  while (i < m && j < n) {
    if (left[i] < right[j]) {
      // 如果左边数组的值小于右边数组的值
      result.push(left[i++]) // 将左边数组的值放入结果数组中
    } else {
      // 否则将右边结果的值放入结果数组中
      result.push(right[j++])
    }
  }
  // 退出循环 判断左右数组是不是有生意
  while (i < m) {
    result.push(left[i++])
  }
  while (j < n) {
    result.push(right[j++])
  }
  return result // 返回合并后的数组
}
let arr = [1, 8, 9, 5, 10, 3, 6, 2, 8, 21, 13]
console.log(sortArray(arr));
42.5快速排序 

/**
 * @param {number[]} nums
 * @return {number[]}
 */
var sortArray = function (nums) {
  quickSort(nums)
  return nums
};
function quickSort (nums, left, right) {
  let len = nums.length
  left = typeof left === 'undefined' ? 0 : left
  right = typeof right === 'undefined' ? len - 1 : right
  if (left < right) {
    let partition = getPartition(nums, left, right)
    quickSort(nums, left, partition - 1)
    quickSort(nums, partition + 1, right)
  }
}
function getPartition (nums, left, right) {
  let pivot = left
  let index = pivot + 1
  for (let i = index; i <= right; i++) {
    if (nums[i] < nums[pivot]) {
      swap(nums, i, index)
      index++
    }
  }
  swap(nums, pivot, index - 1)
  return index - 1
}
function swap (nums, left, right) {
  [nums[left], nums[right]] = [nums[right], nums[left]]
}

<script>
        var items = document.querySelectorAll(".item");//图片节点
        var points = document.querySelectorAll(".point")//点
        var left = document.getElementById("leftBtn");
        var right = document.getElementById("rightBtn");
        var all = document.querySelector(".wrap")
        var index = 0;
        var time = 0;//定时器跳转参数初始化
        
 
        //封装一个清除active方法
        var clearActive = function () {
            for (i = 0; i < items.length; i++) {
                items[i].className = 'item';
            }
            for (j = 0; j < points.length; j++) {
                points[j].className = 'point';
            }
        }
 
        //改变active方法
        var goIndex = function () {
            clearActive();
            items[index].className = 'item active';
            points[index].className = 'point active'
        }
        //左按钮事件
        var goLeft = function () {
            if (index == 0) {
                index = 4;
            } else {
                index--;
            }
            goIndex();
        }
 
        //右按钮事件
        var goRight = function () {
            if (index < 4) {
                index++;
            } else {
                index = 0;
            }
            goIndex();
        }
        
 
        //绑定点击事件监听
        left.addEventListener('click', function () {
            goLeft();
            time = 0;//计时器跳转清零
        })
 
        right.addEventListener('click', function () {
            goRight();
            time = 0;//计时器跳转清零
        })
 
        for(i = 0;i < points.length;i++){
            points[i].addEventListener('click',function(){
                var pointIndex = this.getAttribute('data-index')
                index = pointIndex;
                goIndex();
                time = 0;//计时器跳转清零
            })
        }
        //计时器轮播效果
        var timer;
        function play(){
         timer = setInterval(() => {

                goRight();
                time = 0;

        all.onmousemove = function(){
            clearInterval(timer)
        }

// 传入的参数是一个函数和时间
function newInterval (fn, time) {
  // 在内部定义一个函数 这样可以在内部的settime中执行这个函数
  function inside () {
    fn()
    // 后面的执行都是隔一段时间执行
    setTimeout(inside, time)
  }
  // 让第一次执行是隔一段时间执行
  setTimeout(inside, time)
}

46 实现大数相加

首先将两个数字转成字符串,然后不断的弹出,只要两个有一个还有数字,就不断的弹出,相加,需要一个变量保存计算结果

将计算结果的个位提取出来,然后加上结果

然后提取进位

最后,将头部的0给剔除

function sumBingNumber (a, b) {
  // 首先用字符串的方式来保存大数，这样数字在数学表示上就不会发生变化
  let res = ''
  // 初始化res，temp来保存中间的计算结果，并将两个字符串转化为数组，以便进行每一位的加法运算
  let temp = 0
  a = a.split('')
  b = b.split('')
  // 将两个数组的对应的位进行相加，两个数相加的结果可能大于10，所以可能要进位，对10进行取余操作，将结果保存在当前位
  while (a.length || b.length || temp) {
    temp += ~~a.pop() + ~~b.pop()
    res = (temp % 10) + res // 最后一位相加的结果对10取余拼接在结果的前面 因为是从后向前计算
    temp = temp > 9 // 判断当前位是不是大于9,判断是否有进位,若是就将temp复制为true加法中会转换成1 不是就是false 加法中转换成0 // 一直重复上述操作知道结束
  }
  return res.replace(/^0+/, '')
}
console.log(sumBingNumber('000000009999999999999999999999999999999999', '99999999999999999999999999999999999'));
