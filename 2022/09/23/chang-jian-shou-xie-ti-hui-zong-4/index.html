31.实现new

new有四步

创建一个新对象

让新对象的__proto__指向构造函数的prototype

调用构造函数修改this为新对象

判断新的对象是不是引用类型,如果是返回引用类型,不是返回obj,注意要使用instanceof而不是null,是引用类型,typeof不能判断null 


function Animal (type) {
  this.type = type
  // return {'this':'你好'}
}
Animal.prototype.say = function () {
  console.log('狗是' + this.type + '类型');
}
function mockNew () {
  const constructor = [].shift.call(arguments)
  let obj = {}
  obj.__proto__ = constructor.prototype
  // console.log(arguments);
  let ret = constructor.apply(obj, arguments)
  return ret instanceof Object ? ret : obj
}
// let dog = new Animal('dog')
let dog = mockNew(Animal, 'dog')
console.log(dog.type);
dog.say()

32.实现instanceof

判断右边的原型是不是在左边的原型链上


// instanceof 就是判断构造函数的 prototype 属性是否出现在实例的原型链上。
function Animal (type) {
  this.type = type
}
function Dog (type) {
  Animal.call(this, type)
}
function Miao (tyep) {
  this.type = tyep
}
Dog.prototype = Object.create(Animal.prototype)
Dog.prototype.constructor = Dog
let dog = mockNew(Dog, 'dog')
let miao = new Miao('你好')
console.log(myInstanceof(dog, Dog));
console.log(myInstanceof(miao, Dog));
function mockNew () {
  const constructor = [].shift.call(arguments)
  let obj = {}
  obj.__proto__ = constructor.prototype
  let ret = constructor.apply(obj, arguments)
  return ret instanceof Object ? ret : obj
}
function myInstanceof (left, right) {
  let proto = left.__proto__
  while (true) {
    if (proto === null) return false
    if (proto === right.prototype) return true
    proto = proto.__proto__
  }
}

33.实现object.create

是创建一个新的对象,让新对象的__proto__指向后面的值

两种方法


// Object.create()方法创建一个新对象，使用现有的对象来提供新创建的对象的__proto__。
function Animal (name) {
  this.name = name
}

Object.create1 = function (proto, propertyObject = undefined) {
  if (typeof proto !== 'object' && typeof proto !== 'function') {
    // 如果传入的第一个对象不是object 不是 function 就报错
    throw new TypeError('Object prototype may only be an Object or null.')
  }
  if (propertyObject == null) {
    // 如果传入的第二个地下是null 也要报错
    new TypeError('Cannot convert undefined or null to object')
  }
  let obj = {}
  obj.__proto__ = proto
  if (propertyObject !== undefined) {
    Object.defineProperties(obj, propertyObject)
  }
  return obj
}
Object.create2 = function (proto, propertyObject = undefined) {
  // function F () { }
  // F.prototype = proto
  // return new F()
  function F () { }
  F.prototype = proto
  return new F()
}

}

  console.log('你好');
}
Animal.prototype.say = function () {
  console.log(1);
}

34.实现Object.assign

Object.assign() 方法将所有可枚举（Object.propertyIsEnumerable() 返回 true）的自有（Object.hasOwnProperty() 返回 true）属性从一个或多个源对象复制到目标对象，返回修改后的对象

用法

const target = { a: 1, b: 2 };
const source = { b: 4, c: 5 };

const returnedTarget = Object.assign(target, source);

console.log(target);
// expected output: Object { a: 1, b: 4, c: 5 }

console.log(returnedTarget === target);
// expected output: true

代码如下


// 实现 Object.assign 这个函数就是实现浅拷贝
let obj = {
  name: "xiaolidu",
  age: 20,
  hobby: ['吃饭', '睡觉', '打豆豆'],
  power: {
    sleep: '8小时',
    eat: "20个馒头"
  }
}
Object.assign2 = function (targrt, ...source) {
  if (targrt === null) {
    throw new TypeError('Cannot convert undefined or null to object')
  }
  let ret = Object(targrt)
  source.forEach(function (obj) {
    if (obj !== null) {
      for (let key in obj) {
        if (obj.hasOwnProperty(key)) {
          ret[key] = obj[key]
        }
      }
    }
  })
  return ret
}

37.遍历对象所有属性

for in遍历的是自身+原型链上的可枚举的属性

object.keys遍历的是自身的可枚举的属性

object.getOwnPropertyName遍历的是自身可枚举和不可枚举的属性

代码如下


let obj = {
  name: 'xiaolidu',
  age: 20,
  hobbies: {
    sports: ['游泳', '洗澡', '睡觉'],
    music: ['lost-river', 'jinitaimei']
  },
  birthday: new Date('2020/12/20 20:30:30'),
  say: function () {
    console.log('我的名字是' + this.name);
  },
}
Object.defineProperty(obj, 'power', {
  value: '吃饭',
  enumerable: false
})
function People () {
  this.power = '吃饭'
}
obj.__proto__ = new People()
// console.log(obj);
console.log('自身可枚举', Object.keys(obj)); // [ 'name', 'age', 'hobbies', 'birthday','say' ] // 只是得到自身的可枚举的属性
console.log('自身可枚举不可枚举', Object.getOwnPropertyNames(obj)); // 也是没有继承的属性
let result = []
for (let key in obj) {
  result.push(key) // 这样的话就会有继承的属性了
}
console.log('自身和原型链可枚举', result);
// 遍历对象的所有属性
result.forEach((item) => {
  console.log(obj[item]);

let arr = [1, 2, 3, 4, 5, 6]
arr.sort(() => Math.random() - 0.5) // 对数组随机打乱

// 另一种思路,遍历到第一个产生一个随机数,让于正随机数互换,遍历到第二个产生一个随机数,这个随机数不能是1然后互换,以此类推

39.实现冒泡排序

代码如下


// 比较相邻的元素。如果第一个比第二个大，就交换他们两个。

// 对每一对相邻元素做同样的工作，从开始第一对到结尾的最后一对。在这一点，最后的元素应该会是最大的数。

// 针对所有的元素重复以上的步骤，除了最后一个。

// 持续每次对越来越少的元素重复上面的步骤，直到没有任何一对数字需要比比较
    // 编写方法，实现冒泡
function bubble(arr){     // 1:套一个函数的壳子，将参数传入
//外层循环，控制趟数，每一次找到一个最大值
    for (var i = 0; i < arr.length - 1; i++) {
        // 内层循环,控制比较的次数，并且判断两个数的大小
        for (var j = 0; j < arr.length - 1 - i; j++) {
            // 白话解释：如果前面的数大，放到后面(当然是从小到大的冒泡排序)
            if (arr[j] > arr[j + 1]) {
                var temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
 
    }
return arr  //2: 将执行完的结果返回就可以
}
var arr = [29,45,51,68,72,97]; 
console.log(bubble(arr));//[2, 4, 5, 12, 31, 32, 45, 52, 78, 89]

40.随机选取0~100的数加入数组并排序 

function xuanqupaixu(n){
  let result = []
  for(let i = 0;i < n;i++){
    let num = Math.floor(Math.random() * 100)
    result.push(num)
  }
  for(let i = 0;i < n;i++){
    for(let j = 0;j < n - i;j++){
      if(result[j] > result[j+1]){
        [result[j],result[j+1]] = [result[j+1],result[j]]
      }
    }
  }
  return result
}
console.log(xuanqupaixu(10));
