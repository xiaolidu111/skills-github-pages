实现JSONP

定义一个函数，专门用来实现jsonp，传入一个对象，包含url，params，callbackName，这三个属性

函数内部生成一个script表桥，通过url和params拼接处请求路径字符串

设置src为这个script的src属性

将这个script标签添加到页面中


<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>

<body>
  <script>
    window.getUserInfo = function (data) {
      console.log(data);
    }
    const jsonp = function ({ url, params, callbackName }) { // 传入一个对象 有url,params和回调函数名字
      return new Promise((resolve, reject) => {
        const scriptSrc = generateUrl()
        const scriptEle = document.createElement('script')
        scriptEle.src = scriptSrc
        document.body.appendChild(scriptEle)
        resolve()
      })
      function generateUrl () {
        let dataSrc = [] // 解析params对象 生成= & ? 的字符串拼接上callbackName函数名 后端需要返回这个的callbackName(result) result是json数据格式的
        for (let key in params) {
          dataSrc.push(`${key}=${params[key]}`)
        }
        dataSrc = url + "?" + dataSrc.join('&') + `&callbackName=${callbackName}`
        return dataSrc
      }
    }
    jsonp({
      url: 'http://localhost:3000/api/jsonp',
      params: { name: '方现伟', password: 123456 },
      callbackName: 'getUserInfo'
    })
  </script>
</body>

</html>

22.手写ajax

实现思路,创建一个ajax对象函数传入一个参数对象,里面包含方法,网址,参数,和完成后的函数

首先创建xhr对象,将方法转为大写,对参数进行拼接,拼接处url

如果是get方法,调用open方法,将拼接好的加入到url后面,调用send方法

如果是POST方法,调用open方法,设置请求头为application/x-www-form-urlencoded

将进阶好的字符串传到后端

监听onreadystatechange事件,如果状态码是4 就输出这个函数

代码如下


<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>

<body>
  <script>
    function ajax ({
      method, // 方法名
      url, // 请求地址
      params, // 参数
      done // 完成回调函数
    }) {
      let xhr = window.XMLHttpRequest ? new XMLHttpRequest() : new ActiveXObject("Microsoft.XMLHTTP") // 处理兼容性问题
      method = method.toUpperCase() // 转换成大写字母
      // 参数拼接
      let pair = []
      for (let key in params) {
        pair.push(key + '=' + params[key]) // 拼接这个数组
      }
      let str = pair.join('&')
      // 判断请求方法
      if (method === 'GET') {
        xhr.open(method, url + '?' + str) // 拼接路径开始open
        xhr.send()
      }
      if (method === 'POST') {
        xhr.open(method, url) // 先open然后setrequestheader
        xhr.setRequestHeader("content-type", "application/x-www-form-urlencoded")
        xhr.send(str)
      }
      xhr.onreadystatechange = function () {
        if (this.readyState === 4) {
          // open(JSON.parse(xhr.responseText))
          // console.log(JSON.parse(xhr.responseText));
          done(JSON.parse(xhr.responseText))
        }
      }
    }
    // ajax({
    //   method: 'get',
    //   url: 'http://localhost:3000/api/getuser',
    //   params: {
    //     name: '方现伟',
    //     age: '20岁'
    //   },
    //   done: function (params) {
    //     console.log(params);
    //   }
    // })
    ajax({
      method: 'post',
      url: 'http://localhost:3000/api/addUser',
      params: {
        name: '方现伟',
        age: '20岁'

        throw new TypeError('this is null')

      if (typeof callback !== 'function') {
        throw new TypeError('callback is not a function')

27.实现reduce

通常用于数组求和,接受两个参数,一个是函数,一个是起始值

函数也有两个参数,一个是起始值也是后面函数内部的上一次的返回值,第二个是当前值

要注意,遍历的时候碰到undefined需要调过

代码如下


<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
</head>

<body>
  <script>
    let arr = [, 2, , 4, 5]
    let res = arr.reduce(function (sum, item) {
      return sum + item
    })
    console.log(res);
    Array.prototype.reduce2 = function (callback, initialValue) {
      if (this == null) {
        throw new TypeError('this is null or not defined')
      }
      if (typeof callback !== "function") {
        throw new TypeError(callback + ' is not a function')
      }
      let acc = 0 // 结果
      let k = 0 // 第几个
      if (initialValue !== undefined) { // 传递了初始值
        acc = initialValue
      } else { // 没有传递初始值
        // 去数组中第一个不是undefined的作为初始值
        while (k < this.length) {
          if (this[k]) {
            acc = this[k++]
            break // 退出循环此时的k是第一个是值的位置 所以要k++从下一位开始累加
          }
          k++
        }
      }
      while (k < this.length) {
        if (this[k]) {
          acc = callback(acc, this[k], k, this)
        }
        k++
      }
      return acc
    }
    res = arr.reduce2(function (sum, item) {
      return sum + item
    })
    console.log(res);

  </script>
</body>

</html>

28.实现call

接受两个参数,一个是this指向,一个是参数序列

首先获得this指向,如果没有就是window

获得参数序列

将原函数挂载到新this上

调用新函数的方法获得返回值

删除this的属性

返回这个函数

代码如下


// 使用一个指定的 this 值和一个或多个参数来调用一个函数。

// 实现要点：

// this 可能传入 null；
// 传入不固定个数的参数；
// 函数可能有返回值；
Function.prototype.myCall = function (context) {
  // context = context || window // context是否存在 不存在就是window
  // context.fn = this // 将该方法添加到传进来的this上
  // let args = [...arguments].slice(1) // 接收传递过来的参数
  // let result = null // 初始化返回值
  // result = context.fn(...args)
  // delete context.fn
  // return result
  context = context || window
  context.fn = this
  let args = [...arguments].slice(1)
  let result = context.fn(...args)
  delete context.fn
  return result
}
function a (a) {
  console.log(this.name);
  console.log(this.age);
  console.log(a);
}
let obj = {
  name: '方现伟',
  age: 20
}
a.myCall(obj, 2)

29.实现apply

方法差不多


Function.prototype.myApply = function (context) {
  context = context || window
  context.fn = this
  let args = [...arguments].slice(1)[0] // 截取之后是数组包着数组 去除这个数组中的第一个元素
  let result = null
  if (args && args.length) {
    result = context.fn(...args)
  } else {
    result = context.fn()
  }
  delete context.fn
  return result
}
