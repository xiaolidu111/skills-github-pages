11 事件总线

事件总线也就是发布订阅模式，可以通过一个类来进行创建

首先构造函数创建一个容器，也就是 cache，用来存放对应的键值对

键名表示被订阅者，值是一个数组，表示所有的订阅者

这个要实现订阅函数的方法、取消时间的方法、发布事件的方法这三个方法

对于订阅方法，传入的参数是被订阅者的名字和函数

如果这个存在就 push 这个函数，不存在就设置为数组

取消的放啊 如果这个不存在，就直接返回

否则找到所有的函数的下标 然后删除这个函数

发布，找到这个名字所关联的任务，然后遍历这个任务进行执行

只有一次 就删除掉这个 key

代码如下

class EventEmitter {
  constructor() {
    this.cache = {} // 存储名字和幻术的对应关系
  }
  on(name, fn) {
    // 订阅函数方法
    if (this.cache[name]) {
      // 有这个时间名
      this.cache[name].push(fn) // 放入函数
    } else {
      this.cache[name] = [fn]
    }
  }
  off(name, fn) {
    // 取消时间方法
    if (this.cache[name] === undefined) return
    let tasks = this.cache[name] // 找到这个task
    let index = tasks.findIndex(f => f === fn || f.callback === fn) // 找个这个的索引
    if (index >= 0) {
      // 删除
      tasks.splice(index, 1)
    }
  }
  emit(name, once = false, ...args) {
    // 发布事件
    if (this.cache[name]) {
      let tasks = this.cache[name].slice() // 拷贝一份改时间
      for (let fn of tasks) {
        // 遍历事件传入参数
        fn(...args)
      }
      if (once) {
        // 如果只有一次 删除该事件
        delete this.cache[name]
      }
    }
  }
}

12.解析 URL 为对象

思路见代码

function parseParam(url) {
  // 接受一个参数为url字符串
  let paramsStr = url.split('?')[1] // 首先用?进行分解 取出?后面的 是键值对对象
  let paramsArr = paramsStr.split('&') // 然后用&进行分解 得出这个每个键值对
  let paramsObj = {} // 键值对对象初始化
  for (let i = 0; i < paramsArr.length; i++) {
    // 遍历键值对数组
    if (paramsArr[i].includes('=')) {
      // 如果有等于号
      let [key, val] = paramsArr[i].split('=') // 分割出key和value
      val = decodeURIComponent(val) // 对val进行字符串转义
      if (paramsObj.hasOwnProperty(key)) {
        // 如果对象中已经存在了这个key
        if (paramsObj[key] instanceof Array) {
          // 如果是数组 就加入到数组中
          paramsObj[key].push(val)
        } else {
          paramsObj[key] = [paramsObj[key], val] // 不是数组就转成数组 然后加入数组中
        }
      } else {
        // 如果kye不存在 就是一个值
        paramsObj[key] = val
      }
    } else {
      paramsObj[paramsArr[i]] = true // 如果没有等于号 那默认是true
    }
  }
  return paramsObj // 返回这个对象
}

13.字符串模板

将对象中的数据替换到字符串中,使用正则来进行匹配

首先定义一个正则表达式 用来匹配{{}}中的字符

调用 test 方法 判断是不是能匹配到 如果能匹配到 调用 exec 返回数组,具体用法如下

reg.exec() 的作用是专门在正则中查找字符串中所有敏感词的内容和位置的函数。找到一个敏感词，就会返回一个数组：

数组:[ 0: “敏感词的内容”, index: 敏感词的下标位置 i ] 

var str = '1ab2zzzzz'
var reg = /\d(\w)(\w)\d/ //注意：每个(\w)代表一个子表达式
var ret = reg.exec(str)
console.log(ret[0]) //1ab2
console.log(ret[1]) //a
console.log(ret[2]) //b

如果带括号的话就是子表达式,而且是在匹配内容的后面进行排序

代码如下

function render(template, data) {
  const reg = /\{\{(\w+)\}\}/ // 模板字符串正则
  if (reg.test(template)) {
    // 判断模板里是否有模板字符串
    const name = reg.exec(template)[1] // 查找当前模板里第一个模板字符串的字段
    template = template.replace(reg, data[name]) // 将第一个模板字符串渲染
    return render(template, data) // 递归的渲染并返回渲染后的结构
  }
  return template // 如果模板没有模板字符串直接返回
}

15.函数防抖

概念很清晰

function debounce(fn, time) {
  let timer
  return function () {
    let context = this
    let args = [...arguments]
    if (timer) clearTimeout(timer)
    timer = setTimeout(() => {
      fn.apply(context, args)
    }, time)
  }
}

16.复杂版函数防抖

可以控制立即执行,也可以取消

function debounce(fn, wait, immediate) {
  let timer, result
  let debounceFn = function () {
    let context = this
    let args = arguments
    if (timer) clearTimeout(timer)
    if (immediate) {
      let callNow = !timer // 第一次点击此时的timer为空
      timer = setTimeout(() => {
        // 后续的点击timer就有值了 怎么点击都不会调用这个函数
        timer = null
      }, wait)
      if (callNow) result = fn.apply(context, args)
    } else {
      timer = setTimeout(() => {
        fn.apply(context, args)
      }, wait)
    }
    return result
  }
  debounceFn.cancel = function () {
    if (timer) clearTimeout(timer)
    timer = null
  }
  return debounceFn
}

17.函数节流

概念很清晰

function throttle(fn, time) {
  let timer = null
  return function () {
    let context = this
    let args = [...arguments]
    if (!timer) {
      timer = setTimeout(() => {
        fn.apply(context, args)
        timer = null
      }, time)
    }
  }
}

19.函数科里化

实现传入一个函数作为参数,传出一个可以连续调用的函数 

function curry(fn) {
  // 接受一个函数作为参数
  // console.log(fn.length); fn.length 就是这个函数的参数的长度
  let judge = (...args) => {
    // args此时是一个数组 判断这个数组的长度是不是和fn.legnth的长度相同
    if (fn.length <= args.length) {
      // 如果两者长度相同 那就直接返回这个函数的调用就好
      return fn(...args)
    } // 如果长度不相同 函数调用的时候 就将这个函数的参数拼接到返回的函数参数中
    return (...arg) => judge(...args, ...arg)
  }
  return judge
}
