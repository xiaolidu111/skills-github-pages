回溯算法解题套路框架

解决一个回溯问题,实际上就是一个决策树遍历的过程,站在回溯树的一个节点上,只需要思考三个问题

  • 路径:也就是已经做出的选择
  • 选择列表: 也就是当前可以做的选择
  • 结束条件:也就是决策树的底层,无法再做出的选择的条件 
回溯算法的框架

var backTracking = function(路径,选择列表) {
    if(满足结束条件){
        加入路径
        return
    }
    for(选择 of 选择列表){
        做选择
        backTracking(路径,选择列表)
        撤销选择
    }
};
其实重点就会里面的for循环,在递归调用之前做选择,在调用之后撤销选择

全排列问题

46. 全排列

思路:全排列不允许出现重复的数字,比如1,2,3这三个数字,可以选择先选择了2,那这个2就是路径,1,3就是选择列表,结束条件就是到了叶子节点也就是path的长度和nums的长度相同

这个backTracking函数就是一个指针,在这棵树上游走,同时要维护每个节点的属性,每当走到树的叶子节点其路径就是一个全排列
所以这一题,需要的是 在地柜之前做出选择,在递归之后撤销选择,同时因为不能使用同一个元素,所以需要一个used来记录这个节点是否出现
代码如下 


var permute = function(nums) {
    let result = []
    let path = []
    let used = []
    function backTracking(){
        if(path.length === nums.length){
            result.push([...path])
            return
        }
        for(let i = 0;i < nums.length;i++){
            if(used[i]) continue
            used[i] = true
            path.push(nums[i])
            backTracking()
            used[i] = false
            path.pop()
        }
    }
    backTracking()
    return result
};

回溯算法的时间复杂度走在O(N!)左右,因为回溯算法本质是穷举,复杂度很高

51. N 皇后

思路:
主要的思路就是回溯,通过每一行来进行计算 ,然后再对每一列的元素尝试使用皇后填充,如果是有效的,那就回溯下一个,如果无效就continue
判断这个是否有效 主要就是左上角 右上角 同一列这三个方位 如果出现了不同的就返回false
可以使用set来进行判断,如果是左上角 那行号-列号应该相同 如果是右上角name行号+列号应该相同 然后在用一个集合判断列号是否相同就行了

排除那些不符合条件的分支。提高程序的运行效率。

回溯：

找到答案：返回结果，尝试其他的分支

找不到答案：返回上一层，尝试其他分支

        // 撤销选择 重置状态

}

/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
    // 采用递归，终止条件是字符串的长度等于2n，递归函数传入构建的字符串，左右括号剩余多少，每个位置有两种选择，选择左或者右括号，这里可以进行剪枝优化，只有右括号的保有数量大于左括号的保有数量，才能选右括号，否则肯定不能构成有效括号
    let result = []
    function dfs(str,left,right){
        if(str.length === 2 * n){
            result.push(str) // 如果这个字符串的长度是2n 就加入result
            return 
        }
        if(left > 0){
            // 只要是左括号有剩余,就可以选择做扩哈,继续递归的做选择
            dfs(str + '(',left-1,right) // 左括号的长度-1
        }
        if(right > left){
            // 只有当右括号大于左括号的大小,才可以选择
            dfs(str + ')',left,right - 1)
        }
    }
    dfs('',n,n)
    return result
};

/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
    if(n === 0) return []
    let result = []
    let tarck = []
    function backTracking(left,right){ // 传入参数左右括号剩余数量
        if(left < 0 || right < 0){
            return  // 如果不合法就return
        }
        if(left > right){
            return // 左括号剩下的比右括号多也是不合法的
        }
        if(left === 0 && right === 0){
            // 左右都是剩下0个了 那就说到了最终的了
            result.push(tarck.join('')) // 加入到结果中
            return 
        }
        tarck.push('(') // 先放左括号
        backTracking(left-1,right)
        tarck.pop() // 弹出左括号
        tarck.push(')')
        backTracking(left,right-1)
        tarck.pop()
    }
    backTracking(n,n)
    return result
};

/**
 * @param {number} n
 * @return {string[][]}
 */
 var solveNQueens = function(n) {
  const isValid = function(row,col,chessBoard,n){
    // 判断每一列有没有重复的
    for(let i = 0;i < row;i++){
      if(chessBoard[i][col] === 'Q'){
        return false
      }
    }
    // 判断对角线45°们不能是Q
    for(let i = row - 1,j = col - 1;i >= 0 && j >= 0 ;i--,j--){
      if(chessBoard[i][j] === 'Q'){
        return false
      }
    }
    for(let i = row - 1,j = col + 1;i >= 0 && j < n;i--,j++){
      if(chessBoard[i][j] === 'Q'){
        return false
      }
    }
    return true
  }
    const transformChessBoard = function(chessBoard){
      let chessBoardBack = [] // 定义结果数组
      chessBoard.forEach(row => { // 遍历棋盘的每一行
        let rowStr = '' // 生成一个空字符串
        row.forEach(value => { // 遍历每一行的元素
          rowStr += value // 让每一行的元素加上响应的值 将每一行的元素形成一个字符串
        })
        chessBoardBack.push(rowStr) // 放入结果数组中
      })
      return chessBoardBack // 返回结果数组
    }
    let result = []
    const backTracking = function(row,chessBoard){
      // 传入参数 第几行 棋盘
      if(row === n){
        // 如果传入的行数等于传入的值n 也就是到达了最底部
        result.push(transformChessBoard(chessBoard)) // 将棋盘插入到结果数组中
        return
      }
      for(let col = 0;col < n;col++){
        // 遍历每一行的每一个元素
        if(isValid(row,col,chessBoard,n)){ // 传入的是当前的行 当前的列 和棋盘元素 和n
          // 如果满足列不相等对角线不相等的条件
          chessBoard[row][col] = 'Q'
          backTracking(row+1,chessBoard) // 进入下一次递归
          chessBoard[row][col] = '.' // 退出递归进入修改成默认值.
        }
      }
    }
    
    let chessBoard = new Array(n).fill([]).map(()=>new Array(n).fill('.')) // 生成默认棋盘
    backTracking(0,chessBoard)
    return result
};

        board[i] = new Array(n).fill('.');//生成board
    }

    const cols = new Set();  // 列集，记录出现过皇后的列
    const diag1 = new Set(); // 正对角线集
    const diag2 = new Set(); // 反对角线集
    const res = [];//结果数组

    const backtrack = (row) => {
        if (row == n) {//终止条件
            const stringsBoard = board.slice();
            for (let i = 0; i < n; i++) {//生成字符串
                stringsBoard[i] = stringsBoard[i].join('');
            }

            return;

        for (let col = 0; col < n; col++) {
            // 如果当前点的所在的列，所在的对角线都没有皇后，即可选择，否则，跳过
            if (!cols.has(col) && !diag1.has(row + col) && !diag2.has(row - col)) {
                board[row][col] = 'Q';  // 放置皇后
                cols.add(col);          // 记录放了皇后的列
                diag2.add(row - col);   // 记录放了皇后的正对角线
                diag1.add(row + col);   // 记录放了皇后的负对角线
                backtrack(row + 1);
                board[row][col] = '.';  // 撤销该点的皇后
                cols.delete(col);       // 对应的记录也删一下
                diag2.delete(row - col);
                diag1.delete(row + col);
            }
        }
    };
    backtrack(0);
    return res;
};

/**
 * @param {character[][]} board
 * @return {boolean}
 */
var isValidSudoku = function(board) {
    // 主要思路,用一个对象记录列 一个对象记录行 一个对象记录小方格 
    // 小方格的序号可以按照 parseInt((i/3)) * 3 + parseInt(j/3)来计算
    let rows = {}
    let cols = {}
    let boxs = {} // 分别是行列小方格
    for(let i = 0;i < 9;i++){
        for(let j = 0;j < 9;j++){
            if(board[i][j] !== '.'){ // 如果不是这个.的话说明是个有效的数字
                let num = board[i][j] // 记录这个有效数字
                if(rows[i+'-'+num] || cols[j+'-'+num] || boxs[parseInt(i/3)*3 + parseInt(j/3)+'-'+num]){
                    return false // 这几个有一个是真的 就返回false
                }
                rows[i+'-'+num] = true // 前面的i记录行 后面的num是数字 如果后面还有就是false
                cols[j+'-'+num] = true
                boxs[parseInt(i/3)*3 + parseInt(j/3)+'-'+num] = true // 分别是行列和小方格出现过
            }
        }
    }
    return true // 便利完成就返回true
};

var solveSudoku = function(board) {
    function isValid(row, col, val, board) {
        let len = board.length
        // 行中的数字不能重复
        for(let i = 0; i < len; i++) {
            if(board[row][i] === val) {
                return false
            }
        }
        // 列中的数字不能重复
        for(let i = 0; i < len; i++) {
            if(board[i][col] === val) {
                return false
            }
        }
        let startRow = Math.floor(row / 3) * 3
        let startCol = Math.floor(col / 3) * 3

        //方块中的数字不能重复
        for(let i = startRow; i < startRow + 3; i++) {
            for(let j = startCol; j < startCol + 3; j++) {
                if(board[i][j] === val) {
                    return false
                }
            }
        }

        return true
    }

    function backTracking() {//回溯函数
        for(let i = 0; i < board.length; i++) {
            for(let j = 0; j < board[0].length; j++) {//循环行和列
                if(board[i][j] !== '.') continue
                for(let val = 1; val <= 9; val++) {//尝试在当前单元格放置1-9
                    if(isValid(i, j, `${val}`, board)) {//判断放置数字的合法性
                        board[i][j] = `${val}`//放置数字
                        if (backTracking()) {//合法返回ture
                            return true
                        }
                        
                        board[i][j] = `.`//不合法回溯状态
                    }
                }
                return false//1-9的数字都不合法，返回false
            }
        }
        return true//全部可能性都尝试完成 返回true 说明有解
    }
    backTracking()
    return board
    
};

/**
 * @param {character[][]} board
 * @param {string} word
 * @return {boolean}
 */
var exist = function(board, word) {
    let m = board.length
    let n = board[0].length
    let visited = new Array(m).fill(0).map(()=>new Array(n).fill(false))
    let dirs = [[1,0],[-1,0],[0,1],[0,-1]]
    function backTracking(i,j,k){
        if(board[i][j] !== word[k]){
            return false
        }else if(k === word.length-1){
            return true
        }
        visited[i][j] = true // 表示以及访问过了
        for(let dir of dirs){
            let ni = i + dir[0]
            let nj = j + dir[1] // 新的坐标
            if(ni < 0 || ni >= m || nj < 0 || nj >= n){
                continue
            }
            if(!visited[ni][nj]){ // 如果没被访问过 才进行下一步
                if(backTracking(ni,nj,k+1)){
                    return true // 如果下一个是true 那这个就返回true
                }
            }
        }
        // 如果都不是真的 就让这个设置为未访问过
        visited[i][j] = false
        return false // 返回没找到
    }
    for(let i = 0;i < m;i++){
        for(let j = 0;j < n;j++){
            if(backTracking(i,j,0)){
                return true
            }
        }
    }
    return false
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
 var permute = function(nums) {
    let res = [], path = [], used = []
    // used 是记录第几位用没用
    const backTracking = function(){
      if(path.length === nums.length){
        res.push([...path])
        return
      }
      for(let i = 0;i < nums.length;i++){
        if(used[i]) continue // 如果当前下标的元素已经在同一个树枝被使用过久调过这个循环
        path.push(nums[i])
        used[i] = true
        backTracking() // 下一次继续遍历
        path.pop()
        used[i] = false 
      }
    }
    backTracking()
    return res
};

/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
 var combine = function(n, k) {
    let res = [] // 结果数组 储存这个回溯得到的结果
    let path = [] // 储存这个的路径
    // 定义回溯的函数
    // 参数 n,k和这个遍历的起始数字 返回值就是没有
    // 退出 当让path的大小等于k的时候 就退出
    // 单层逻辑 遍历 放入path数组 回溯函数调用 弹出上次插入的数字
    const backTracting = function(n,k,startIndex){
      if(path.length === k){
        res.push([...path])
        return        
      }
      for(let i = startIndex;i <= n;i++){ 
        path.push(i)
        backTracting(n,k,i+1)
        path.pop()
      }
    }
    backTracting(n,k,1)
    return res
};

var letterCombinations = function(digits) { // 这个是数字的字符串
    // 首先定义一个map数组结构
    let map = ['','','abc','def','ghi','jkl','mon','pqrs','tuv','wxyz']
    // 需要一个字符串作为遍历得到的字符串 还需要一个结果数组
    // 遍历每一个数字对应的字符串 然后组成新的字符串 当遍历数字字符串到达结尾 就push到结果数组
    // 参数 需要传入的参数是
    let path = [],res = []
    let len = digits.length
    const backTracking = function(index){ // index表示遍历到第几个数字
        if(path.length === len){
            // 如果字符串的长度就是数字字符串的长度 就表示成功了一个 加入结果数组
            res.push(path.join(''))
            return
        }
        let str = map[digits[index]] // index代表遍历到第几个数字
        for(let i = 0;i < str.length;i++){
            path.push(str[i]) // 遍历到的字母加入路径
            backTracking(index+1)
            path.pop()
        }
    }
    if(len === 0) return res
    backTracking(0)
    return res
};

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
 var subsets = function(nums) {
    let res = []
    let path = []
    const backTracking = function(startIndex){
      // 每次以进入函数就把path加入到结果数组中
      res.push([...path])
      // 退出条件 当startindex 大于等于这个nums的长度的时候就退出
      if(startIndex >= nums.length){
        return
      }
      for(let i = startIndex;i < nums.length;i++){
        path.push(nums[i]) // 将当前的元素加入到path中 下一步就加入到结果数组
        backTracking(i+1) // 下一步从i+1开始计算 计入结果数组中
        path.pop() // 上面的到底之后就弹出 进入下一个i的循环
      }
    }
    backTracking(0)
    return res
};

/**
 * @param {number[]} matchsticks
 * @return {boolean}
 */
var makesquare = function(matchsticks) {
    let sum = matchsticks.reduce((p,val)=>p+val) // 求和
    if(matchsticks.length < 4) return false // 没有四个也直接返回
    if(sum % 4) return false // 如果除不尽4 就直接返回false
    matchsticks.sort((a,b)=>b-a) // 降序排序
    let bucket = new Array(4).fill(0) // 准备四个空桶
    let result = backTracking(0,sum/4) // 计算传入这个元素的下标,和每个桶的最大容量
    function backTracking(i,edge){ // i表示传入的下标  edge表示每个桶最大的容量
        if(i >= matchsticks.length){ // 如果这个超出了桶的范围 就直接返回true
            return true
        }
        // 遍历每个桶
        for(let j = 0;j < 4;j++){
            if(bucket[j] + matchsticks[i] > edge){
                // 如果这个桶的容量加上当前的值大于这个最大容量 就进行下一个桶的判断
                continue
            }
            bucket[j] += matchsticks[i] // 加上这个桶
            if(backTracking(i+1,edge)){
                // 不断遍历下一个桶,如果最红返回的true 说明可以成功
                return true
            }
            // 如果不成功 就回溯一下,这个位置不能放这个数字
            bucket[j] -= matchsticks[i]
        }
        // 如果遍历完都没有成功 就直接返回false
        return false
    }
    return result
};
