62. 不同路径 

/**
 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var uniquePaths = function(m, n) {
       // dp数组含义 dp[i][j]表示走到ij位置的最多的路径条数
       // 地推公式 dp[0][j] = dp[0][j-1] dp[i][0] = dp[i-1][0] 
       // dp[i][j] = dp[i-1][j] + dp[i][j-1]
       // 初始化 dp[0] = 1 
       // 遍历顺序 从左到右 从上到下
       let dp = new Array(m).fill(0).map(()=>new Array(n).fill(1))
       for(let i = 0;i < m;i++){
               for(let j = 0;j < n;j++){
                       if(i === 0 && j === 0){
                               // ij都是0 那就初始化为1
                               dp[i][j] = 1
                       }else if(i === 0){
                               // ij有一个不是0 那就这个就是这个的前一个或者是左边或者是右边
                               dp[i][j] = dp[i][j-1]
                       }else if(j === 0){
                               dp[i][j] = dp[i-1][j]
                       }else{
                               // 如果都是不是0 那就是左边+上面
                               dp[i][j] = dp[i-1][j] +dp[i][j-1]
                       }
               }
       }
       return dp[m-1][n-1]
};

 * @param {number} m
 * @param {number} n
 * @return {number}
 */
var uniquePaths = function(m, n) {
       // 一维数组解决 
       let dp = new Array(n)
       for(let i = 0;i < n;i++) dp[i] = 1 // 初始化第一行的数组的最短路径都是1
       for(let i = 1;i < m;i++){ // 从第2行判断
               for(let j = 1;j < n;j++){ // 从第二列判断
                       dp[j] = dp[j-1] + dp[j] // 当前的值就是之前的路线 加上 前面一位的路线
               }
       }
       return dp[n-1] // 最后返回最后一个的路线
};

/**
 * @param {number[][]} obstacleGrid
 * @return {number}
 */
var uniquePathsWithObstacles = function(obstacleGrid) {
       // dp[i][j]表示到这个位置的路径数
       // 递推公式如果这个不是障碍 那么就是左边的加上右边的 如果这个是障碍 那么这个就是0
       // 这样的话下一个的取值就是左边的加上面的 其中有一个是障碍的话就是单独的一个了
       let m = obstacleGrid.length
       let n = obstacleGrid[0].length
       let dp = new Array(m).fill(0).map(()=>new Array(n).fill(0)) // 第二这个ob数组是二维数组 要分清行列
       for(let i = 0;i < n && obstacleGrid[0][i] !== 1;i++){
               dp[0][i] = 1 // 将dp数组的上边界和左边界都赋值为1 除了obs数字该数字是1的情况第一不能让dp[0][0]是1 因为要避免[1](onenote:#1&section-id={AE715DAD-B069-4885-A89B-1727F809BE27}&page-id={B7A4B202-E19F-48C8-B6A3-ABA0CFF9A838}&end&base-path=https://onedriveplus-my.sharepoint.cn/personal/x76_onedriveplus_live/Documents/代码随想录练习题/动态规划.one)这种情况
       }
       for(let i = 0;i < m && obstacleGrid[i][0] !== 1;i++){
               dp[i][0] = 1
       }
       for(let i = 1;i < m;i++){
               for(let j = 1;j < n;j++){
                       if(obstacleGrid[i][j] === 1) continue // 如果这个数字是个障碍 那么就调过这个数字 让其还是0
                       dp[i][j] = dp[i-1][j] +dp[i][j-1] // 不是障碍 就是左边的加上上边的
               }
       }
       return dp[m-1][n-1]
};

343. 整数拆分

两重循环 外层循环用来确定这个dp[i]的i下标,内层循环用来从1到n一直判断
公式就是 dp[i] = math.max(dp[i],jdp[i-j],i(i-j))
==注意:不要忘了j*(i-j)== 

/**
 * @param {number} n
 * @return {number}
 */
var integerBreak = function(n) {
       // dp[i]表示i的最大拆分乘积
       // dp[i] 是什么相当于 从1开始拆到i-1假设拆掉的值是j 比较j*(i-j) 和j*dp[i-j]的最大值 第一个相当于单独拆 第二个相当于将i-j继续拆分
       // 初始化初始化dp[2] = 1
       // 遍历顺序 从左享有
       let dp = new Array(n+1).fill(0)
       dp[1] = 1
       dp[2] = 1
       for(let i = 3;i <= n;i++){
               // 可以遍历到n 这个是可以取值n dp[n]
               for(let j = 1;j < i;j++){
                       // 从j=1遍历到j = n-1
                       dp[i] = Math.max(dp[i],dp[i-j]*j,(i-j)*j) // 这里为什么还包含dp[i] 就是在运动计算的过程中dp[i]是不断变化的 在内层循环中不断变化
               }
       }
       console.log(dp)
       return dp[n]
};

96.不同的二叉搜索树

分成左右两个子树来看,左子树的个数可以是0~i-1个 因为 根元素肯定要有,那么当这个左子树的个数从0变化到i-1的时候 右子树元素的个数是从i-1变化到0,所以当我们求dp[i]也就是i个元素的二叉树的个数的时候,就可以让左子树乘以右子树不断的相加得到结果 

/**
 * @param {number} n
 * @return {number}
 */
var numTrees = function(n) {
       // dp[i]表示1-i个数字组成的搜索二叉树的个数
       // 递推公式 让j 从0到i dp[i] = dp[j]*dp[i-j] 表示的意思就是左子树是0-j组成的二叉搜索树的个数
       // 乘右子树是以i-j组成的二叉搜索是的个数 凑是从0开始 因为有可能左右子树是null
       // 初始化dp【0】 = 1 这个是为了相乘有值
       // 递归顺序 从左到右
       let dp = new Array(n+1).fill(0) // 不能初始化为1 这样就会后面相加有个初始值 
       dp[0] = 1 // 初始化dp[0]是1
       for(let i = 1;i <= n;i++){ // 从第2位开始遍历一致到第n为 这一行是头结点 从1到n
               for(let j = 0;j < i;j++){ // 下面从j = 1开始遍历 这一行是从左边是0遍历到左边是全部 取不到i就不会-1了
                       dp[i] += dp[j]*dp[i-j-1] // 左子树是0-i-1的二叉搜索树个数乘以右子树是0-i-1的二叉搜索树个数 所以说可以等于i 这里的j-1是为了让其能从0开始到j-1 后面也是可以从i-1到0
               }
       }
       console.log(dp)
       return dp[n]
};

322. 零钱兑换

完全背包问题,因为是这个硬币可以用无数次
也是两重for循环,外层遍历背包,内层遍历物品
dp[i]的含义是凑够i的钱的最小的硬币用量
那么这个dp[i]就可以由硬币的面值来推出
==dp[i] = Math.min(dp[i],dp[i-coin]+1)==
那么外层遍历物品内层遍历背包行不行呢
事实证明也是可以的 

/**
 * @param {number[]} coins
 * @param {number} amount
 * @return {number}
 */
// 外层遍历背包 内层遍历物品
var coinChange = function(coins, amount) {
       let dp = new Array(amount+1).fill(Infinity)
       dp[0] = 0
       for(let i = 1;i <= amount;i++){
               for(let coin of coins){
                       if(coin <= i){
                               dp[i] = Math.min(dp[i],dp[i-coin]+1)
                       }
               }
       }
       return dp[amount] === Infinity ? -1 :dp[amount]
};
// 外层遍历物品 内层遍历背包

 * @param {number[]} coins
 * @param {number} amount
 * @return {number}

var coinChange = function(coins, amount) {
       let dp = new Array(amount+1).fill(Infinity)
       dp[0] = 0
       for(let coin of coins){
               for(let j = coin;j <= amount;j++){
                       dp[j] = Math.min(dp[j],dp[j-coin]+1)
               }
       }
       return dp[amount] === Infinity ? -1 : dp[amount]
};

10. 正则表达式匹配

主要思路 就是从前向后判断是否匹配

  • 首先是按照两个字符串相比较的规律,生成一个二维的数组
  • dp[i][j]代表的是i-1,j-1的位置的字符是否匹配

首先是初始化

  • 00 就是true 空字符串是互相匹配的
  • 0j代表有长度的匹配空串
    • 如果有长度的是 *的话有可能匹配
    • 其他情况就没办法匹配了
  • 然后就是两个字符串从前向后不断的进行比较
    • 如果两个位置的字符串是相匹配的,那么就说明这里的匹配信息是前面的匹配信息
    • 如果两个字符串后面的模式串是*说明有可能匹配 
/**
 * @param {string} s
 * @param {string} p
 * @return {boolean}
 */
var isMatch = function(s, p) {
       if(s === null || p === null) return false
       const sLen = s.length
       const pLen = p.length
       const dp = new Array(sLen+1)
       for(let i = 0;i < dp.length;i++){
               dp[i] = new Array(pLen+1).fill(false)
       }
       dp[0][0] = true
       for(let j = 1;j < pLen + 1;j++){ // 一个时候'' 一个是字符串
               if(p[j-1] === '*') dp[0][j] = dp[0][j-2] // 如果是*的话要向前看两位
       }
       for(let i = 1;i < sLen + 1;i++){
               for(let j = 1;j < pLen + 1;j++){
                       if(s[i-1] === p[j-1] || p[j-1] === '.'){ // 如果这一位匹配上了
                               dp[i][j] = dp[i-1][j-1] // 向前看一位
                       }else if(p[j-1] === '*'){ // 如果这个是* 
                               if(s[i-1] == p[j-2] || p[j-2] == '.'){ // 1 并且前一位匹配了 那就可能是重复了 有可能重复一次 两次和n次  重复0次 就是dp[i][j-2] 重复n次 就是dp[i-1][j]
                                       dp[i][j] = dp[i][j-2] || dp[i-1][j]
                               }else{ // 如果这个没有匹配完成 当前的状态就是 前两个的状态
                                       dp[i][j] = dp[i][j-2]
                               }
                       }
               }
       }
       return dp[sLen][pLen] // 返回最后的状态
};



/**
 * @param {string} s
 * @param {string} p
 * @return {boolean}
 */
var isMatch = function(s, p) {
       let m = s.length
       let n = p.length
       let dp = new Array(m+1).fill(0).map(()=>new Array(n+1).fill(false)) // 先假定都不能匹配 dp[i][j] 表示i-1,j-1的是否匹配
       dp[0][0] = true // 空字符串是匹配的
       for(let i = 1;i <= m;i++){
               dp[i][0] = false // 有字符串和没有的匹配是false 
       }
       for(let j = 1;j <= n;j++){
               if(p[j-1] === '*'){
                       dp[0][j] =  dp[0][j-2]// 没有字符串和有字符串的匹配 能匹配上吗
               }else{
                       dp[0][j] =  false
               }
               // dp[0][j] =  false// 没有字符串和有字符串的匹配 能匹配上吗
       }
       for(let i = 1;i <= m;i++){
               for(let j = 1;j <= n;j++){
                       if(s[i-1] === p[j-1] || p[j-1] === '.' ){
                               // 如果能匹配完成
                               dp[i][j] = dp[i-1][j-1]
                       }else if(p[j-1] === '*'){
                               if(s[i-1] === p[j-2] || p[j-2] === '.'){
                                       dp[i][j] = dp[i][j-2] || dp[i-1][j]
                               }else{
                                       dp[i][j] = dp[i][j-2]
                               }
                       }else{
                               dp[i][j] = false
                       }
               }
       }
       // console.table(dp)
       return dp[m][n]
};

312. 戳气球

首先在这个气球的两边加上两个1 这样就能够计算戳破两边的情况了
然后生成一个m+2*m+2的dp数组 ==dp[i][j]==表示在i,j这个开区间内最大的硬币数量
那么这个最大的硬币数量就是==I,j==最后戳破 保留I,j中间的位置 之间的乘积的最大值
==dp[i][k] + dp[k][j] + score[j] * score[k] * score[i]==
遍历顺序

  • 从下向上
  • 从左到右
  • k也是从左到右 从i+1到j
/**
 * @param {number[]} nums
 * @return {number}
 */
var maxCoins = function(nums) {
    let score = [1,...nums,1] // 得到的奖品
    let m = nums.length // 获得这个数组的长度
    let dp = new Array(m+2).fill(0).map(()=>new Array(m+2).fill(0)) // dp[i][j] 表示在i和j这个开区间内最大的硬币数量
    for(let i = m;i >= 0;i--){
        for(let j = i+1;j < m + 2;j++){ 
            for(let k = i+1;k < j;k++){
                dp[i][j] = Math.max(
                    dp[i][j],
                    dp[i][k] + dp[k][j] + score[j] * score[k] * score[i] // I,j是最后弄破的,所以这个的大小是k*i:j 左右的是左右区间的值                )
            }
        }
    }
    return dp[0][m+1] 
};

198. 打家劫舍

典型的动态规划,一个房间可以有两种选择,偷还是不偷,可以定义一个二维的数组,0代表不偷1代表偷 这样以此类推来得到相应的公示 

/**
 * @param {number[]} nums
 * @return {number}
 */
var rob = function(nums) {
       let m = nums.length
       let dp = new Array(m).fill(0).map(()=>new Array(2).fill(0))
       dp[0][0] = 0
       dp[0][1] = nums[0]
       for(let i = 1;i < m;i++){
               dp[i][0] = Math.max(dp[i-1][1],dp[i-1][0]) // 前一天偷了 和前一天没偷今天保持
               dp[i][1] = Math.max(dp[i-1][0]+nums[i]) // 前一天没偷 今天偷了
       }
       return Math.max(dp[m-1][1],dp[m-1][0])
};
