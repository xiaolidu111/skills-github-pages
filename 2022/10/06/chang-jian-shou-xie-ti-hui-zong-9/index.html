83.小孩报数问题

代码如下


function childNum (num, count) {
  let allplayer = []; // 生成一个所有的人的数组
  for (let i = 0; i < num; i++) {
    allplayer[i] = i + 1; // 给每个人进行编号
  }

  let exitCount = 0;    // 离开人数
  let counter = 0;      // 记录报数
  let curIndex = 0;     // 当前下标

  while (exitCount < num - 1) { // 当退出的人数小于人数-1 就剩下一个人
    if (allplayer[curIndex] !== 0) counter++; // 退出的人就置为0 然后一次报数 如果不是0 就让报数+1

    if (counter == count) { // 如果报数是要的数 
      allplayer[curIndex] = 0; // 设置的当前的变成0 
      counter = 0; // 计数设为0
      exitCount++; //离开的人数++
    }
    curIndex++; // 当前的下标++
    if (curIndex == num) { // 如果走到末尾
      curIndex = 0 // 修改当前的下标为0
    };
  }
  for (i = 0; i < num; i++) { // 剩下的这一个 寻找不是0的返回
    if (allplayer[i] !== 0) {
      return allplayer[i]
    }
  }
}

85.实现发布订阅模式 


class EventCenter {
  // 1. 定义事件容器，用来装事件数组
  constructor() {
    this.handler = {}
  }

  // 2. 添加事件方法，参数：事件名 事件方法
  addEventListener (type, handler) {
    // 创建新数组容器
    if (!this.handlers[type]) {
      this.handlers[type] = []
    }
    // 存入事件
    this.handlers[type].push(handler)
  }

  // 3. 触发事件，参数：事件名 事件参数
  dispatchEvent (type, params) {
    // 若没有注册该事件则抛出错误
    if (!this.handlers[type]) {
      return new Error('该事件未注册')
    }
    // 触发事件
    this.handlers[type].forEach(handler => {
      handler(...params)
    })
  }

  // 4. 事件移除，参数：事件名 要删除事件，若无第二个参数则删除该事件的订阅和发布
  removeEventListener (type, handler) {
    if (!this.handlers[type]) {
      return new Error('事件无效')
    }
    if (!handler) {
      // 移除事件
      delete this.handlers[type]
    } else {
      const index = this.handlers[type].findIndex(el => el === handler)
      if (index === -1) {
        return new Error('无该绑定事件')
      }
      // 移除事件
      this.handlers[type].splice(index, 1)
      if (this.handlers[type].length === 0) {
        delete this.handlers[type]
      }
    }
  }
}

    function arr2tree (nodes) {
      let result = []
      let map = {}
      nodes.forEach((item) => {
        map[item.id] = item
      })
      nodes.forEach((item) => {
        let parent = map[item.parentId]
        if (parent) {
          if (parent.children === undefined) {
            parent.children = []
          }
          parent.children.push(item)
        } else {
          result.push(item)
        }
      })
      return result
    }
    function tree2arr (tree) {
      let result = []
      tree.forEach((item) => {
        if (item.children === undefined) {
          result.push(item)
        } else {
          let obj = {}
          for (let key in item) {
            if (item.hasOwnProperty(key) && key !== 'children') {
              obj[key] = item[key]
            }
          }
          let childrenItems = tree2arr(item.children)
          result.push(...tree2arr(childrenItems))
          result.push(obj)
        }
      })
      return result
    }

88.取消promise

设置一个函数,传入一个要取消的promise对象

函数内部,生成一个空对象,再生成一个新的promise对象,将这个新对象的resolve,和reject方法挂载到这个promise对象上,然后使用promise.race方法计算这个新的promise对象和要取消的promise对象,最后将这个promise对象返回

可以在函数外部接收这个promise对象,调用这个的resolve方法完成取消 

// 定义一个promise对象
let promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve(123)
  }, 3000);
})
// 封装一个取消promise的函数,使用promise.race的特性
function stopPromise (stopP) { // 传入promise对象
  let proObj = {} // 空对象
  let promise = new Promise((resolve, reject) => {
    proObj.resolve = resolve
    proObj.reject = reject // 创建一个新的promise对象保存这个resolve和reject函数
  })
  proObj.promise = Promise.race([stopP, promise]) // 让这个的promise属性是race的值
  return proObj // 返回这个对象
}
let obj = stopPromise(promise) // 电泳函数返回一个对象
obj.promise.then(res => { // 收集返回值
  console.log(res);
})
setTimeout(() => {
  obj.resolve('promise 请求被取消了')
}, 2000);

90.原生ajax获取上传进度

首先是放下一个input输入框,设置type=file,然后给这个输入框绑定onchange事件,监听该事件,通过dom实例获得到这个里面的file文件对象,然后利用这个生成一个formdata对象,将这个文件添加到formdata对象中,调用xhr的方法进行上传,监听xhr.upload.onprogresss这个事件,可以通过事件对象的loaded属性获得已经上传的大小,通过total获得总大小计算出这个的百分比,完成上传显示

代码如下


<!DOCTYPE html>
<html lang="en">

<head>
  <meta charset="UTF-8">
  <meta http-equiv="X-UA-Compatible" content="IE=edge">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>Document</title>
  <style>
    .box {
      position: relative;
      width: 200px;
      height: 20px;
      background-color: skyblue;
      border-radius: 10px;
      line-height: 20px;
      padding: 0 1px;
      transition: all 0.1s ease;
    }

    .text {
      position: absolute;
      z-index: 999;
      width: 100%;
      color: white;
      text-align: center;
    }

    .inner {
      position: absolute;
      border-radius: 10px;
      width: 10%;
      height: 20px;
      background-color: red;
    }
  </style>
</head>

<body>
  <input type="file" name="" id="fileId" onchange="changeFn()">
  <div class="box">
    <div class="inner"></div>
    <div class="text">0%</div>
  </div>
  <script>
    function changeFn () {
      let fileDOM = document.querySelector('#fileId')
      let inner = document.querySelector('.inner')
      let text = document.querySelector('.text')
      console.log(fileDOM.files[0]); // 是这个的文件列表 FileList
      let formdata = new FormData()
      formdata.append('file', fileDOM.files[0])
      let xhr = new XMLHttpRequest()
      xhr.open('POST', 'http://localhost:3000/api/addUser')
      xhr.onreadystatechange = function () {
        if (xhr.readyState === 4) {
          console.log(xhr.responseText);
        }
      }
      let percent = 0
      xhr.upload.onprogress = function (event) {
        if (event.lengthComputable) {
          percent = Math.floor(event.loaded / event.total * 100)
          console.log(percent);
          inner.style.width = percent + '%'
          text.innerText = percent + '%'
        }
      }
      xhr.send(formdata)
    }

  </script>
</body>

</html>
